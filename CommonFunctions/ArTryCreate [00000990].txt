	SetFunctionValue aArray
elseif eval iType ==0
	SetFunctionValue ar_Construct "Array"

	SetFunctionValue ar_Construct "Map"
else ;if eval iType ==2
	SetFunctionValue ar_Construct "Stringmap"
endif
end
